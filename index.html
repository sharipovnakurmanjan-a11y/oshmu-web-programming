HTML / Глава 01— 1 / 20 —HTML Негиздери
Глава 01 · HTML
HTML Негиздери
HyperText Markup Language — веб-баракчанын скелети жана тиреги
— ✦ —

HTML — HyperText Markup Language деп толук ачылат. Аны 1991-жылы британиялык окумуштуу Тим Бернерс-Ли ойлоп тапкан. Бүгүнкү күндө интернеттеги бардык сайттар HTML негизинде иштейт. HTML — программалоо тили эмес, ал белгилөө тили (markup language).

📌 Маанилүү

HTML браузерге мазмунду кантип көрсөтүүнү айтат. Азыркы учурда HTML5 версиясы колдонулат — ал видео, аудио, геолокация, canvas жана башка заманбап мүмкүнчүлүктөрдү колдойт.

1.1 HTML документинин структурасы

Ар бир HTML файл бир стандарттуу скелетке ээ. Бул скелет браузерге документти туура окуу мүмкүнчүлүгүн берет:

HTML — index.html 
<!-- HTML5 декларациясы -->
<!DOCTYPE html>
<html lang="ky">
 
  <head>
    <!-- Мета маалымат -->
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>Менин биринчи сайтым</title>
    <link rel="stylesheet" href="style.css">
  </head>
 
  <body>
    <!-- Колдонуучуга көрүнүүчү мазмун -->
    <h1>Саламатсызбы, Дүйнө!</h1>
    <p>Бул менин биринчи HTML барагым.</p>
    <script src="script.js"></script>
  </body>
 
</html>

Глава 05 · CSS
CSS Негиздери
Cascading Style Sheets — кооз дизайндын тили
— ✦ —

CSS — Cascading Style Sheets, баскычтама стиль барактары. HTML мазмундун структурасын берсе, CSS аны визуалдык жактан кооздойт: түстөр, шрифттер, өлчөмдөр, жайгашуулар жана анимациялар.

5.1 CSS Синтаксиси
CSS — style.css
/* Базалык CSS структурасы */
/* СЕЛЕКТОР { КАСИЕТ: МААНИ; } */
 
/* 1. Элемент селектору */
body {
  font-family: Georgia, 'Times New Roman', serif;
  background-color: #fdf6ed;
  color: #2d1a06;
  margin: 0;
  padding: 0;
}
 
h1 {
  font-size: 48px;
  font-weight: bold;
  color: #3d1f0a;
  text-align: center;
  margin-bottom: 24px;
}
 
/* 2. Класс селектору — . менен */
.баскыч {
  background: #c8922a;
  color: white;
  padding: 12px 28px;
  border: none;
  border-radius: 6px;
  cursor: pointer;
  font-size: 16px;
  transition: all 0.3s ease;
}
 
.баскыч:hover {
  background: #8b4513;
  transform: translateY(-2px);
}
 
/* 3. ID селектору — # менен */
#башкы-меню {
  position: fixed;
  top: 0;
  width: 100%;
  background: #3d1f0a;
  z-index: 1000;
}
 
/* 4. Мурастык (Descendant) */
.карточка p {
  font-size: 14px;
  color: #5c3a1a;
}
 
/* 5. CSS Variables (Касиет өзгөрмөлөрү) */
:root {
  --негизги-түс: #c8922a;
  --текст-түс: #2d1a06;
}
 
.баскыч {
  background: var(--негизги-түс);
}

Flexbox жана Grid
Заманбап веб-layoutтун эки мощный куралы

7.1 Flexbox
CSS — Flexbox
.контейнер {
  display: flex;
 
  /* Горизонталдуу тегиздөө */
  justify-content: center;
  /* flex-start | flex-end | center | space-between | space-around | space-evenly */
 
  /* Вертикалдуу тегиздөө */
  align-items: center;
  /* flex-start | flex-end | center | stretch | baseline */
 
  /* Багыт */
  flex-direction: row;
  /* row | row-reverse | column | column-reverse */
 
  flex-wrap: wrap;
  gap: 16px;
}
 
.бала {
  flex: 1;        /* Калган орунду бөлүп ал */
  flex: 0 0 200px; /* Фиксирленген туурасы */
  order: 2;        /* Орунду өзгөрт */
}
7.2 CSS Grid
CSS — Grid
.сетка {
  display: grid;
 
  /* 3 тилке, бирдей туурасы */
  grid-template-columns: repeat(3, 1fr);
 
  /* Жооптуу тилкелер */
  grid-template-columns: repeat(auto-fit, minmax(250px, 1fr));
 
  /* Катарлар */
  grid-template-rows: 100px auto 80px;
 
  gap: 24px;
}
 
/* Аймактар аркылуу */
.орун {
  grid-template-areas:
    "баш   баш   баш"
    "ченеп мазм  каптал"
    "аяк   аяк   аяк";
}
 
.баш-бөлүм  { grid-area: баш; }
.мазмун     { grid-area: мазм; }
.аяк-бөлүм  { grid-area: аяк; }

CSS — Анимация жана Media Query
/* TRANSITION */
.баскыч {
  transition: all 0.3s ease;
}
.баскыч:hover {
  transform: scale(1.05);
  box-shadow: 0 8px 24px rgba(0,0,0,0.2);
}
 
/* KEYFRAMES АНИМАЦИЯ */
@keyframes пайда-болуу {
  from {
    opacity: 0;
    transform: translateY(30px);
  }
  to {
    opacity: 1;
    transform: translateY(0);
  }
}
 
.карточка {
  animation: пайда-болуу 0.6s ease forwards;
  animation-delay: 0.2s;
}
 
/* MEDIA QUERY — Жооптуу дизайн */
/* Телефондор */
@media (max-width: 600px) {
  .контейнер {
    flex-direction: column;
    padding: 16px;
  }
  h1 { font-size: 28px; }
}
 
/* Планшеттер */
@media (max-width: 900px) {
  .сетка {
    grid-template-columns: repeat(2, 1fr);
  }
}
 
/* Принтер стили */
@media print {
  .навигация, .рекламалар { display: none; }
}

// Класс аныктоо
class Адам {
  // constructor — объект жараткандагы метод
  constructor(ат, жашы) {
    this.ат = ат;
    this.жашы = жашы;
  }

  тааныштыр() {
    return `${this.ат}, ${this.жашы} жашта`;
  }

  // Статикалык метод — объекттен эмес, класстан чакырылат
  static жардам() {
    return "Адам классы ат жана жашты сакталат";
  }
}

// Мурастоо (Inheritance)
class Студент extends Адам {
  constructor(ат, жашы, университет) {
    super(ат, жашы); // Ата-энеден мурастоо
    this.университет = университет;
  }

  тааныштыр() {
    return super.тааныштыр() + `, ${this.университет} студенти`;
  }
}

// Объект жаратуу
const айзат = new Студент("Айзат", 21, "ОшМУ");

Веб-тиркемелер сервер менен маалымат алмашат — бул операция убакыт талап кылат. JavaScript'те асинхрондуу код — Promise, async/await жана Fetch API — менен серверден маалымат алабыз.

17.1 Promise (Убада)
JavaScript — Promise
// Promise 3 абалда болот:
// pending (күтүүдө), fulfilled (ийгиликтүү), rejected (ката)

const убада = new Promise((resolve, reject) => {
  const ийгилик = true;
  if (ийгилик) {
    resolve("Маалымат алынды!");
  } else {
    reject("Ката болду!");
  }
});

// .then() / .catch() менен иштетүү
убада
  .then(натыйжа => console.log(натыйжа))
  .catch(ката => console.error(ката))
  .finally(() => console.log("Бүттү"));

// Бир нече Promise бир убакта
Promise.all([убада1, убада2, убада3])
  .then(натыйжалар => console.log(натыйжалар));
17.2 Async/Await + Fetch API
JavaScript — Async/Await + Fetch
// async функция — await колдонуу үчүн
async function колдонуучуларАлуу() {
  try {
    // API'ден маалымат алуу
    const жооп = await fetch(
      'https://jsonplaceholder.typicode.com/users'
    );

    // Статусту текшерүү
    if (!жооп.ok) {
      throw new Error(`Ката: ${жооп.status}`);
    }

    // JSON форматка айландыруу
    const маалымат = await жооп.json();
    console.log(маалымат);
    return маалымат;

  } catch (ката) {
    console.error("Ката болду:", ката);
  }
}

// POST жиберүү
async function маалыматЖибер(маалымат) {
  const жооп = await fetch('/api/users', {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify(маалымат)
  });
  return await жооп.json();
}

// Функцияны чакыруу
колдонуучуларАлуу();

LocalStorage — браузерде маалымат сактоонун оңой жолу. Бет жабылса да, компьютер өчсө да маалымат сакталып кала берет. Колдонуучунун тандоолорун, темасын, тиркелгисин сактоо үчүн идеалдуу.

18.1 LocalStorage API
JavaScript — LocalStorage
// ЖАЗУУ — setItem(ачкыч, маани)
localStorage.setItem('колдонуучу_аты', 'Айзат');
localStorage.setItem('тема', 'dark');

// Объект сактоо (JSON.stringify керек!)
const студент = { ат: "Айзат", упай: 95 };
localStorage.setItem('студент', JSON.stringify(студент));

// ОКУУ — getItem(ачкыч)
const ат = localStorage.getItem('колдонуучу_аты');
console.log(ат); // 'Айзат'

// Объектти окуу (JSON.parse керек!)
const сакталган = JSON.parse(
  localStorage.getItem('студент')
);

// ЖОК КЫЛУУ
localStorage.removeItem('тема');  // Бирди жок кыл
localStorage.clear();              // Баарын тазала

// ТЕКШЕРҮҮ
if (localStorage.getItem('тема') === 'dark') {
  document.body.classList.add('dark-mode');
}

19.2 JavaScript логикасы
JavaScript — To-Do Logic
// LocalStorage'тен жүктөө
let тизме = JSON.parse(
  localStorage.getItem('милдеттер')
) || [];

function кошуу() {
  const input = document.getElementById('жаңы-милдет');
  const текст = input.value.trim();
  if (!текст) return;

  тизме.push({ id: Date.now(), текст: текст, аткарылдыбы: false });
  input.value = '';
  сакта();
  чийип_бер();
}

function аткарылды(id) {
  const милдет = тизме.find(т => т.id === id);
  if (милдет) милдет.аткарылдыбы = !милдет.аткарылдыбы;
  сакта();
  чийип_бер();
}

function жок_кыл(id) {
  тизме = тизме.filter(т => т.id !== id);
  сакта();
  чийип_бер();
}

function сакта() {
  localStorage.setItem('милдеттер', JSON.stringify(тизме));
}

function чийип_бер() {
  const ul = document.getElementById('тизме');
  ul.innerHTML = тизме.map(т => `
    <li class="${т.аткарылдыбы ? 'аткарылды' : ''}">
      <input type="checkbox" ${т.аткарылдыбы?'checked':''} 
             onchange="аткарылды(${т.id})">
      <span>${т.текст}</span>
      <button onclick="жок_кыл(${т.id})">✕</button>
    </li>
  `).join('');
  const калган = тизме.filter(т => !т.аткарылдыбы).length;
  document.getElementById('калган').textContent = `${калган} милдет калды`;
}
чийип_бер(); // Баштоодо чакыр
